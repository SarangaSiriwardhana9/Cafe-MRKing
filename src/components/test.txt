import React from 'react';
import { Link } from 'react-router-dom';
import { useState } from 'react';
import { FaHeart } from 'react-icons/fa';

const Cards = ({ item }) => {
  const [isHeartFilled, setIsHeartFilled] = useState(false);

  const handleHeartClick = () => {
    setIsHeartFilled(!isHeartFilled);
  };

  return (
    <div className="card w-72 h-78 bg-white rounded-lg  shadow-2xl overflow-hidden">
      <div className="relative">
        <div
          className={`rating absolute  top-2 right-2 p-2 rounded-full bg-green ${
            isHeartFilled ? 'text-rose-500' : 'text-white'
          }`}
        >
          <FaHeart className="h-4 w-4 cursor-pointer" onClick={handleHeartClick} />
        </div>
        <Link to={`/menu/${item._id}`}>
          <div className="flex justify-center items-center h-48"> {/* Center image horizontally and vertically */}
            <img
              src={item.image}
              alt=""
              className="object-cover w-48 h-full sm:h-48 hover:scale-105 transition-all duration-200"
            />
          </div>
        </Link>
      </div>

      <div className="p-4">
        <h2 className="text-xl font-semibold">{item.name}</h2>
        <p className="text-gray-500 mt-2">{item.description}</p>
        <div className="flex justify-between items-center mt-4">
          <h5 className="text-red-500 font-semibold">{item.price}</h5>
          <button className="btn bg-green text-white hover:bg-hovergreen">
            Buy Now
          </button>
        </div>
      </div>
    </div>
  );
};

export default Cards;
